4    * Clase que despliega la solucion a una secuencia de nodos en consola.

8   public class Solucion {
9   	/**
10  	 * Instancia que construye una coleccion de nodos que están conextados
11  	 * secuencialmente por medio de su propiedad next.
12  	 * 
13  	 * @return Node Regresa el primero que se creó.
14  	 */
15  	private Node build() {
16  	    Node n1 = new Node("A");
17  	    Node n2 = new Node("B");
18  	    Node n3 = new Node("C");
19  	    Node n4 = new Node("D");
20  	    Node n5 = new Node("E");
21  	    Node n6 = new Node("F");
22  	    n1.setNext(n2);
23  	    n2.setNext(n3);
24  	    n3.setNext(n4);
25  	    n4.setNext(n5);
26  	    n5.setNext(n6);
27  	    return n1;
28  	}
29  
30      /**
31       * Prueba la funcionalidad de el método "gus" con distintos casos de prueba.
32       */
33      public void ok() {
34          Node a = build();
35          a.prn();
36          a = a.gus();
37          a.prn();
38          a = a.gus();
39          a.prn();
40          
41          Node b = new Node("X");
42          b = b.gus();
43          b.prn();
44          
45          Node c = new Node("X");
46          Node d = new Node("Y");
47          c.setNext(d);
48          c = c.gus();
49          c.prn();   
50      }
